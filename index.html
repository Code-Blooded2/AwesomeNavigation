Awesome Nav keeps both screens alive during transitions — enabling parallel animations, predictive back gestures, and per-screen lifecycle that actually works.

⚡
Z-Stack Rendering
Every entry renders simultaneously. No AnimatedContent overhead. Both screens alive during transition.
🧠
Scoped ViewModels
Each NavEntry owns its ViewModelStore. VMs are cleared only after the exit animation completes — never before.
🔄
Per-Entry Lifecycle
ON_RESUME and ON_PAUSE fire correctly per screen. Background screens don't keep collecting flows.
🔒
Type-Safe Routes
Pure Kotlin sealed classes. No NavType registration, no string IDs, no URI conversion at runtime.
🌐
Deep Links
Single addDeepLink() pattern registration. Pending queue for cold-start links. No NavDeepLink per destination.
📡
NavRegistry
Access the navigator from anywhere — ViewModels, Services, BroadcastReceivers — without passing it around.

Define Routes

All routes must be a @Serializable sealed class implementing NavRoute.

kotlin
@Serializable
sealed class AppRoute : NavRoute {
    @Serializable data object Home    : AppRoute()
    @Serializable data object Settings: AppRoute()
    @Serializable data class  Detail(val id: Int, val title: String) : AppRoute()
}
Setup NavHost
kotlin
@Composable
fun App() {
    val navigator = rememberNavigator<AppRoute>(AppRoute.Home)

    NavHost(
        navigator       = navigator,
        backgroundColor = MaterialTheme.colorScheme.background
    ) { route ->
        when (route) {
            is AppRoute.Home     -> HomeScreen()
            is AppRoute.Settings -> SettingsScreen()
            is AppRoute.Detail   -> DetailScreen(route.id, route.title)
        }
    }
}
Navigate
kotlin
// Push
navigator.push(AppRoute.Detail(id = 42, title = "Item"))

// Back
navigator.back()

// Replace current screen
navigator.replace(AppRoute.Settings)

// Clear stack and set new root
navigator.clearAll(AppRoute.Home)

// Pop up to a route
navigator.popUpTo(AppRoute.Home, inclusive = false)
Animation Styles

Configure once in Application.onCreate().

kotlin
AwesomeNav.setup {
    enterDuration  = 300
    exitDuration   = 250
    animationStyle = NavAnimationStyle.SlideHorizontal // default
    // SlideHorizontal · SlideVertical · Fade · ScaleFade · None
}
Result Passing
kotlin
// Screen B — set result before going back
navigator.setResult("picked_item", selectedItem)
navigator.back()

// Screen A — observe and consume
LaunchedEffect(Unit) {
    navigator.results.collect {
        val item = navigator.consumeTypedResult<Item>("picked_item")
                   ?: return@collect
        // handle result
    }
}
NavRegistry — Access Anywhere
kotlin
// From a ViewModel, Service, or BroadcastReceiver
val navigator = NavRegistry.get<AppRoute>()
navigator?.push(AppRoute.Detail(id = 1, title = "From ViewModel"))
Access Navigator Easily
kotlin
// Anywhere in route
    val navigator = currentNavigator<AppRoute>()

navigator?.push(AppRoute.Detail(id = 1, title = "From ViewModel"))
Deep Links
kotlin
AwesomeNav.setup {
    addDeepLink("myapp://detail/{id}/{title}") { params ->
        AppRoute.Detail(
            id    = params["id"]?.toIntOrNull() ?: return@addDeepLink null,
            title = params["title"]        ?: return@addDeepLink null
        )
    }
}

// In Activity
override fun onCreate(savedInstanceState: Bundle?) {
    super.onCreate(savedInstanceState)
    setContent { App() }
    intent?.let { AwesomeNav.handleDeepLink(it) }
}

override fun onNewIntent(intent: Intent) {
    super.onNewIntent(intent)
    AwesomeNav.handleDeepLink(intent)
}

    kotlin
    dependencyResolutionManagement {
    repositories {
        maven { url = uri("https://code-blooded2.github.io/AwesomeNavigation/") }
    }

    kotlin
    implementation("awesome.navigation:navigation:1.0.0")

    kotlin
    // build.gradle.kts (app)
plugins {

}
